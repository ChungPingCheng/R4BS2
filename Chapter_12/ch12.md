``` r
#重排教育各水準
#重排婚姻，依發生頻次由多到少
#將年齡換成年次
#將婚姻狀況分為未婚與結過婚
dta <- dta |> 
  dplyr::mutate(教育 = factor(教育, 
                              levels = c('國小以下', '國中', '高中', '專科', '大學以上')),
                婚姻 = factor(婚姻, 
                            levels = c('有偶', '未婚', '離婚', '喪偶')),
                年次 = (102.5 - 年齡),
                結過婚 = ifelse(婚姻 == '未婚', 0, 1))
```
