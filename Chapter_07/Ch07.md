## 區辨度

``` r
#計算區辨度。以總分為準，選取低分組與高分組，比較各題在兩組上的差異。
dta_2g <- dta %>% 
  dplyr::mutate( tot = rowSums(.),
                 grp = case_when(
                   rank(tot) > 500*.73 ~ "H",
                   rank(tot) < 500*.27 ~ "L",
  .default = "M")) |>
  dplyr::filter(grp != 'M')
```

``` r
#t檢定, 算高低分組平均數
#將計算結果存於新資料框架中
item_t <- sapply(dta_2g[,1:35], 
                 function(x) tidy(t.test(x ~ dta_2g$grp)))[1:4,] |> 
  t() |> 
  as.data.frame() |>
  dplyr::mutate(across(1:4, as.numeric)) |>
  dplyr::rename(c("組平均差" = "estimate", 
                  "低分組平均"="estimate2", 
                  "高分組平均"="estimate1", 
                  "t檢定值"="statistic"))
```
